class Solution {
    public int findComplement(int num) {
        int count=0;
        int original=num;
        while(num>0){
            count++;
            num=num>>1;
        }
        int mask=(1<<count)-1;
        return original^mask;
        }
        public static void main(String[] args){
            int num=5;
            Solution sol=new Solution();
            System.out.println(sol.findComplement(num));
        }
}
